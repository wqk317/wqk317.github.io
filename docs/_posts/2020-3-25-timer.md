# 倒计时

```
@echo off

::倒计时
set /a i = 5
:Timeout
if %i% == 0 goto Next
setlocal
set /a i = %i% - 1
cls
echo begin with in %i% s...
choice /t 1 /d y /n >nul
goto Timeout
goto End
:Next
::这里添加正式脚本
echo Start!!!
```
